# AI Stock Selection - Cron 定时任务配置
# 用于自动执行数据抓取、分析等任务

# 环境变量
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
NODE_ENV=production

# ============ 定时任务 ============

# 每个交易日开盘前（9:00）- 抓取昨日数据
0 9 * * 1-5 /root/.openclaw/workspace/ai-stock-selection/scripts/dispatch-jobs.sh >> /var/log/ai-stock/cron.log 2>&1

# 每个交易日收盘后（15:30）- 执行分析和复盘
30 15 * * 1-5 /root/.openclaw/workspace/ai-stock-selection/scripts/dispatch-jobs.sh analyze >> /var/log/ai-stock/cron.log 2>&1

# 每天凌晨 2 点 - 清理过期数据
0 2 * * * /root/.openclaw/workspace/ai-stock-selection/scripts/cleanup.sh >> /var/log/ai-stock/cron.log 2>&1

# 每周一早上 8 点 - 发送周报
0 8 * * 1 /root/.openclaw/workspace/ai-stock-selection/scripts/weekly-report.sh >> /var/log/ai-stock/cron.log 2>&1
